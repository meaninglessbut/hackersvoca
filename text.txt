        async function fetchWithRetry(url, options, maxRetries = 3) {
            let delay = 1000;
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    const responseData = await response.json();
                    
                    if (!response.ok) {
                        console.error("API Response Error:", responseData);
                        let errorMessage = `HTTP error! status: ${response.status}`;
                        if (responseData.error && responseData.error.message) {
                            errorMessage += ` - ${responseData.error.message}`;
                        }
                        throw new Error(errorMessage);
                    }
                    return responseData;
                } catch (error) {
                    if (i === maxRetries - 1) throw error;
                    await new Promise(resolve => setTimeout(resolve, delay));
                    delay *= 2;
                }
            }
        }

        async function getBestModel(apiKey) {
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models?key=${apiKey}`);
                if (!response.ok) return 'gemini-3.5-flash-lite'; 
                
                const data = await response.json();
                const models = data.models || [];
                
                const preferred = ['gemini-3.5-flash-lite', 'gemini-1.5-flash-8b', 'gemini-2.5-flash', 'gemini-2.0-flash', 'gemini-1.5-flash'];
                for (const pref of preferred) {
                    const found = models.find(m => m.name.includes(pref) && (m.supportedGenerationMethods || []).includes('generateContent'));
                    if (found) return found.name.replace('models/', '');
                }
                
                const fallback = models.find(m => (m.supportedGenerationMethods || []).includes('generateContent'));
                return fallback ? fallback.name.replace('models/', '') : 'gemini-3.5-flash-lite';
            } catch (error) {
                return 'gemini-3.5-flash-lite';
            }
        }

        async function generateQuizFromWords(selectedWords, allWords, apiKey, questionCount, apiModel) {
            if (selectedWords.length === 0) return null;

            const modelName = apiModel && apiModel.trim() !== '' ? apiModel.trim() : await getBestModel(apiKey);
            const url = `https://generativelanguage.googleapis.com/v1beta/models/${modelName}:generateContent?key=${apiKey}`;
            
            const shuffledWords = [...selectedWords].sort(() => 0.5 - Math.random());
            const targetWords = shuffledWords.slice(0, questionCount);
            
            const targetWordListStr = targetWords.map(w => w.word).join(", ");
            const allWordListStr = allWords.map(w => w.word).join(", ");
            const distractorCount = Math.min(3, allWords.length - 1);

            const prompt = `
                I am an English vocabulary learner. I am taking a quiz on these target words: [${targetWordListStr}].
                My ENTIRE vocabulary pool is: [${allWordListStr}].
                
                Please generate a fill-in-the-blank vocabulary quiz with exactly ${targetWords.length} questions using the target words.
                For each question:
                1. Create a natural English example sentence.
                2. Replace the target word with '_____' in the sentence.
                3. Provide the correct target word.
                4. Provide exactly ${distractorCount} distractors (incorrect options).
                
                CRITICAL INSTRUCTIONS FOR DISTRACTORS: 
                - You MUST choose the distractors ONLY from my ENTIRE vocabulary pool provided above.
                - The distractors MUST NOT be synonyms of the correct answer and MUST NOT fit contextually in the blank. 
                - Ensure there is only ONE uniquely correct answer from the options provided.
            `;

            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: {
                    parts: [{ text: "You are an expert English teacher creating strict vocabulary quizzes. You never provide duplicate answers or use outside words." }]
                },
                generationConfig: {
                    responseMimeType: "application/json",
                    responseSchema: {
                        type: "OBJECT",
                        properties: {
                            questions: {
                                type: "ARRAY",
                                items: {
                                    type: "OBJECT",
                                    properties: {
                                        question: { type: "STRING" },
                                        correctAnswer: { type: "STRING" },
                                        distractors: {
                                            type: "ARRAY",
                                            items: { type: "STRING" }
                                        }
                                    },
                                    required: ["question", "correctAnswer", "distractors"]
                                }
                            }
                        },
                        required: ["questions"]
                    }
                }
            };

            const data = await fetchWithRetry(url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            });

            try {
                const jsonString = data.candidates[0].content.parts[0].text;
                const parsedData = JSON.parse(jsonString);
                
                const allWordsLowerCase = allWords.map(w => w.word.toLowerCase());

                const questionsWithOptions = parsedData.questions.map(q => {
                    let validDistractors = (q.distractors || []).filter(d => 
                        allWordsLowerCase.includes(d.toLowerCase()) && 
                        d.toLowerCase() !== q.correctAnswer.toLowerCase()
                    );

                    // AI가 외부 단어를 넣었다면 JS 로직으로 단어장 풀에서 랜덤하게 채워 넣음
                    if (validDistractors.length < distractorCount) {
                        const usedLowerCase = [q.correctAnswer.toLowerCase(), ...validDistractors.map(v => v.toLowerCase())];
                        const remainingPool = allWordsLowerCase.filter(w => !usedLowerCase.includes(w));
                        
                        for (let i = remainingPool.length - 1; i > 0; i--) {
                            const j = Math.floor(Math.random() * (i + 1));
                            [remainingPool[i], remainingPool[j]] = [remainingPool[j], remainingPool[i]];
                        }
                        
                        const needed = distractorCount - validDistractors.length;
                        validDistractors.push(...remainingPool.slice(0, needed));
                    }

                    validDistractors = validDistractors.slice(0, distractorCount);

                    const options = [q.correctAnswer, ...validDistractors];
                    for (let i = options.length - 1; i > 0; i--) {
                        const j = Math.floor(Math.random() * (i + 1));
                        [options[i], options[j]] = [options[j], options[i]];
                    }
                    return { ...q, options };
                });

                return questionsWithOptions;
            } catch (error) {
                console.error("Parse Error Details:", error);
                throw new Error("Invalid response format from AI.");
            }
        }

                        {/* 스텝 1: 유닛 범위 선택 */}
                        <div className="p-6 border-b border-slate-100">
                            <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-4 gap-3">
                                <h2 className="text-xl font-bold text-slate-800 flex items-center gap-2">
                                    <Folder className="w-5 h-5 text-indigo-500" />
                                    1. 학습 및 퀴즈 범위 선택
                                </h2>
                                <div className="bg-slate-100 p-1 rounded-lg inline-flex">
                                    <button onClick={selectAllCategories} className="px-3 py-1.5 text-xs font-bold text-slate-600 hover:bg-white hover:text-indigo-600 rounded-md transition-all">전체 선택</button>
                                    <button onClick={deselectAllCategories} className="px-3 py-1.5 text-xs font-bold text-slate-500 hover:bg-white hover:text-red-500 rounded-md transition-all">전체 해제</button>
                                </div>
                            </div>
                            
                            <div className="bg-slate-50 border border-slate-200 rounded-xl p-4 max-h-56 overflow-y-auto custom-scrollbar">
                                {availableCategories.length === 0 ? (
                                    <div className="text-center text-slate-400 py-4">저장된 단어 카테고리가 없습니다.</div>
                                ) : (
                                    <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 gap-3">
                                        {availableCategories.map(cat => (
                                            <label key={cat} className={`flex items-center justify-center p-3 rounded-xl border-2 cursor-pointer transition-all select-none ${quizCategorySelection.includes(cat) ? 'bg-indigo-50 border-indigo-500 text-indigo-700 font-bold shadow-sm' : 'bg-white border-slate-200 text-slate-600 hover:border-indigo-200 hover:bg-slate-50'}`}>
                                                <input type="checkbox" checked={quizCategorySelection.includes(cat)} onChange={() => toggleQuizCategory(cat)} className="hidden" />
                                                {cat}
                                            </label>
                                        ))}
                                    </div>
                                )}
                            </div>
                            <div className="mt-3 text-sm text-slate-500 font-medium text-right">
                                선택된 유닛: <span className="text-indigo-600 font-bold">{quizCategorySelection.length}</span>개 
                                (총 <span className="text-indigo-600 font-bold">{words.filter(w => quizCategorySelection.includes(w.category)).length}</span> 단어)
                            </div>
                        </div>

                        {/* 스텝 2: 퀴즈 모드 및 옵션 설정 */}
                        <div className="p-6 bg-slate-50/50">
                            <h2 className="text-xl font-bold text-slate-800 mb-4 flex items-center gap-2">
                                <Settings className="w-5 h-5 text-indigo-500" />
                                2. 퀴즈 상세 옵션
                            </h2>

                            <div className="flex flex-col sm:flex-row gap-4 mb-6">
                                <label className={`flex-1 flex items-center gap-3 p-4 rounded-xl border-2 cursor-pointer transition-all bg-white ${quizMode === 'offline' ? 'border-indigo-500 shadow-sm' : 'border-slate-200 hover:border-indigo-300'}`}>
                                    <input type="radio" name="quizMode" checked={quizMode === 'offline'} onChange={() => setQuizMode('offline')} className="hidden" />
                                    <div className={`w-5 h-5 rounded-full border-2 flex items-center justify-center flex-shrink-0 ${quizMode === 'offline' ? 'border-indigo-500' : 'border-slate-300'}`}>
                                        {quizMode === 'offline' && <div className="w-2.5 h-2.5 bg-indigo-500 rounded-full"></div>}
                                    </div>
                                    <div>
                                        <div className="font-bold text-slate-800 flex items-center gap-1"><WifiOff className="w-4 h-4"/> 오프라인 빠른 퀴즈</div>
                                        <div className="text-xs text-slate-500 mt-0.5">내장된 원래 예문으로 빈칸 퀴즈를 생성합니다.</div>
                                    </div>
                                </label>
                                <label className={`flex-1 flex items-center gap-3 p-4 rounded-xl border-2 cursor-pointer transition-all bg-white ${quizMode === 'online' ? 'border-indigo-500 shadow-sm' : 'border-slate-200 hover:border-indigo-300'}`}>
                                    <input type="radio" name="quizMode" checked={quizMode === 'online'} onChange={() => setQuizMode('online')} className="hidden" />
                                    <div className={`w-5 h-5 rounded-full border-2 flex items-center justify-center flex-shrink-0 ${quizMode === 'online' ? 'border-indigo-500' : 'border-slate-300'}`}>
                                        {quizMode === 'online' && <div className="w-2.5 h-2.5 bg-indigo-500 rounded-full"></div>}
                                    </div>
                                    <div>
                                        <div className="font-bold text-slate-800 flex items-center gap-1"><Sparkles className="w-4 h-4 text-indigo-500"/> AI 응용 문맥 퀴즈</div>
                                        <div className="text-xs text-slate-500 mt-0.5">AI가 새로운 문맥의 예문을 동적으로 만들어냅니다.</div>
                                    </div>
                                </label>
                            </div>

                            {/* 온라인 모드일 때만 나타나는 설정 */}
                            {quizMode === 'online' && (
                                <div className="mb-6 p-5 bg-indigo-50/50 rounded-xl border border-indigo-100 flex flex-col sm:flex-row items-center gap-4 animate-in zoom-in-95 duration-200">
                                    <div className="bg-indigo-100 p-3 rounded-full flex-shrink-0 hidden sm:block">
                                        <Key className="w-5 h-5 text-indigo-500" />
                                    </div>
                                    <div className="flex-1 w-full">
                                        <label className="block text-xs font-bold text-indigo-800 mb-1.5">Gemini API Key</label>
                                        <input type="password" placeholder="AIzaSy..." value={apiKey} onChange={(e) => setApiKey(e.target.value)} className="w-full px-4 py-2.5 rounded-lg border border-indigo-200 focus:outline-none focus:ring-2 focus:ring-indigo-500 text-sm bg-white" />
                                    </div>
                                    <div className="flex-1 w-full">
                                        <label className="block text-xs font-bold text-indigo-800 mb-1.5">API 모델명 (Model)</label>
                                        <input type="text" placeholder="예: gemini-3.5-flash-lite" value={apiModel} onChange={(e) => setApiModel(e.target.value)} className="w-full px-4 py-2.5 rounded-lg border border-indigo-200 focus:outline-none focus:ring-2 focus:ring-indigo-500 text-sm bg-white" />
                                    </div>
                                </div>
                            )}

                            <div className="flex items-center justify-between p-4 bg-white border border-slate-200 rounded-xl">
                                <div className="font-bold text-slate-700">퀴즈 출제 문제 수</div>
                                <div className="flex items-center gap-2">
                                    <input 
                                        type="number" min="3" max={words.filter(w => quizCategorySelection.includes(w.category)).length || 3} 
                                        value={quizQuestionCount} onChange={e => setQuizQuestionCount(Number(e.target.value))} 
                                        className="w-20 px-3 py-2 rounded-lg border border-slate-200 focus:outline-none focus:ring-2 focus:ring-indigo-500 text-center font-bold text-slate-800 bg-slate-50"
                                    />
                                    <span className="text-sm text-slate-500 font-medium">/ 최대 {words.filter(w => quizCategorySelection.includes(w.category)).length}</span>
                                </div>
                            </div>
                        </div>

                        {/* 액션 버튼 */}
                        <div className="p-6 bg-white border-t border-slate-100 flex flex-col sm:flex-row gap-4">
                            <button onClick={startStudy} className="flex-1 py-4 px-6 rounded-2xl font-bold text-lg flex items-center justify-center gap-2 transition-all bg-slate-100 text-slate-700 border border-slate-200 hover:bg-white hover:border-indigo-300 hover:text-indigo-700 hover:shadow-sm">
                                <BookOpen className="w-6 h-6" /> 선택 범위 학습하기
                            </button>
                            <button onClick={startQuiz} className="flex-1 py-4 px-6 rounded-2xl font-bold text-lg flex items-center justify-center gap-2 transition-all bg-gradient-to-r from-indigo-600 to-violet-600 hover:from-indigo-700 hover:to-violet-700 text-white shadow-md hover:shadow-lg transform hover:-translate-y-0.5">
                                <Play className="w-6 h-6 fill-current" /> 선택 범위 퀴즈 시작
                            </button>
                        </div>
                    </div>
                </div>
            );

            // --- 화면 렌더링 2: 단어장 학습 모드 ---
            const renderStudy = () => {
                const filteredWords = words.filter(w => quizCategorySelection.includes(w.category));
                const wordsByCategoryForView = filteredWords.reduce((acc, w) => {
                    if (!acc[w.category]) acc[w.category] = [];
                    acc[w.category].push(w);
                    return acc;
                }, {});

                return (
                    <div className="max-w-4xl mx-auto animate-in fade-in slide-in-from-right-8 duration-500">
                        {/* 상단 컨트롤 바 */}
                        <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4 mb-6 bg-white p-4 rounded-2xl shadow-sm border border-slate-200 sticky top-20 z-40">
                            <div className="flex items-center gap-3">
                                <button onClick={goHome} className="p-2 bg-slate-100 text-slate-600 hover:bg-slate-200 rounded-lg transition-colors">
                                    <ChevronLeft className="w-6 h-6" />
                                </button>
                                <div>
                                    <h2 className="text-xl font-black text-slate-800">선택 범위 학습장</h2>
                                    <p className="text-sm text-slate-500 font-medium">{quizCategorySelection.length}개 유닛 / 총 {filteredWords.length} 단어</p>
                                </div>
                            </div>
                            <button onClick={startQuiz} className="w-full sm:w-auto px-6 py-2.5 rounded-xl font-bold flex items-center justify-center gap-2 transition-all bg-indigo-600 hover:bg-indigo-700 text-white shadow-sm hover:shadow">
                                <Play className="w-4 h-4 fill-current" /> 바로 퀴즈 풀기
                            </button>
                        </div>

                        {/* 단어 리스트 표시 */}
                        <div className="space-y-8">
                            {filteredWords.length === 0 ? (
                                <div className="bg-white p-10 text-center rounded-2xl border border-slate-200 text-slate-500">
                                    선택된 단어가 없습니다. 홈 화면에서 범위를 다시 확인해주세요.
                                </div>
                            ) : (
                                Object.keys(wordsByCategoryForView).sort((a, b) => {
                                    const numA = parseInt(a.replace(/[^0-9]/g, '')) || 0;
                                    const numB = parseInt(b.replace(/[^0-9]/g, '')) || 0;
                                    return numA - numB;
                                }).map(category => (
                                    <div key={category} className="bg-white rounded-3xl shadow-sm border border-slate-200 overflow-hidden">
                                        <div className="bg-slate-50 p-4 border-b border-slate-200">
                                            <h3 className="font-bold text-slate-700 flex items-center gap-2 text-lg">
                                                <Folder className="w-5 h-5 text-indigo-500" />
                                                {category} <span className="text-sm font-normal text-slate-400 bg-white px-2 py-0.5 rounded-full border border-slate-200">{wordsByCategoryForView[category].length} 단어</span>
                                            </h3>
                                        </div>
                                        <div className="divide-y divide-slate-100">
                                            {wordsByCategoryForView[category].map((w) => (
                                                <div key={w.id} className="p-5 sm:p-6 hover:bg-indigo-50/30 transition-colors">
                                                    <p className="font-black text-slate-800 text-2xl mb-2 tracking-tight">{w.word}</p>
                                                    <p className="text-indigo-600 font-bold mb-3">{w.definition}</p>
                                                    <p className="text-slate-600 italic border-l-4 border-slate-200 pl-4 py-1 leading-relaxed text-lg">"{w.example}"</p>
                                                </div>
                                            ))}
                                        </div>
                                    </div>
                                ))
                            )}
                        </div>
                    </div>
                );
            };

            // --- 화면 렌더링 3: 퀴즈 화면 ---
            const renderQuiz = () => {
                if (isGenerating) {
                    return (
                        <div className="flex flex-col items-center justify-center py-32 text-center">
                            <div className="relative mb-8">
                                <div className="absolute inset-0 bg-indigo-200 rounded-full blur-2xl animate-pulse"></div>
                                <Loader2 className="w-20 h-20 text-indigo-600 animate-spin relative z-10" />
                            </div>
                            <h2 className="text-3xl font-black text-slate-800 mb-3">AI가 퀴즈를 출제 중입니다</h2>
                            <p className="text-slate-500 text-lg">선택하신 범위 내에서 교묘한 오답을 만들고 있습니다...</p>
                        </div>
                    );
                }

                if (!quizQuestions.length) return null;

                const currentQ = quizQuestions[currentQuestionIndex];
                const hasAnsweredCurrent = userAnswers.length > currentQuestionIndex;
                const currentAnswer = hasAnsweredCurrent ? userAnswers[currentQuestionIndex].selectedAnswer : null;

                return (
                    <div className="max-w-3xl mx-auto">
                        {/* 퀴즈 진행도 바 */}
                        <div className="mb-10">
                            <div className="flex justify-between items-end mb-2">
                                <button onClick={goHome} className="text-sm font-bold text-slate-400 hover:text-slate-600 flex items-center gap-1 transition-colors">
                                    <ChevronLeft className="w-4 h-4"/> 종료
                                </button>
                                <span className="text-sm font-black text-indigo-600 bg-indigo-50 px-3 py-1 rounded-full border border-indigo-100">
                                    Question {currentQuestionIndex + 1} of {quizQuestions.length}
                                </span>
                            </div>
                            <div className="flex gap-1.5 h-2.5">
                                {quizQuestions.map((_, idx) => (
                                    <div key={idx} className={`flex-1 rounded-full transition-all duration-300 ${idx === currentQuestionIndex ? 'bg-indigo-500 scale-y-125' : idx < currentQuestionIndex ? 'bg-indigo-200' : 'bg-slate-200'}`} />
                                ))}
                            </div>
                        </div>

                        {/* 문제 (예문 문맥 유추) */}
                        <div className="bg-white p-8 sm:p-12 rounded-[2rem] shadow-sm border border-slate-200 mb-8 text-center relative overflow-hidden">
                            {/* 데코레이션 배경 */}
                            <div className="absolute top-0 right-0 p-8 opacity-5">
                                <BookOpen className="w-32 h-32" />
                            </div>
                            
                            <div className="inline-flex items-center justify-center gap-2 px-4 py-2 rounded-xl bg-slate-100 text-slate-600 text-sm font-bold mb-8">
                                {quizMode === 'offline' ? '오프라인 내장 예문' : 'AI 응용 문맥 예문'}
                            </div>
                            
                            <p className="text-2xl sm:text-3xl leading-relaxed text-slate-800 font-bold break-words relative z-10">
                                {currentQ.question}
                            </p>
                            {/* 💡 요청사항 반영: 힌트 박스 완전 제거됨 */}
                        </div>

                        {/* 보기 선택 */}
                        <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                            {currentQ.options.map((option, idx) => {
                                let buttonStateClass = "bg-white border-slate-200 text-slate-700 hover:border-indigo-300 hover:bg-indigo-50 hover:text-indigo-800";
                                let Icon = null;

                                if (hasAnsweredCurrent) {
                                    const isSelected = currentAnswer === option;
                                    const isCorrect = option === currentQ.correctAnswer;
                                    if (isCorrect) {
                                        buttonStateClass = "bg-green-50 border-green-500 text-green-800 shadow-sm ring-4 ring-green-500/20 z-10";
                                        Icon = <CheckCircle className="w-6 h-6 text-green-600" />;
                                    } else if (isSelected && !isCorrect) {
                                        buttonStateClass = "bg-red-50 border-red-400 text-red-800 z-10";
                                        Icon = <XCircle className="w-6 h-6 text-red-500" />;
                                    } else {
                                        buttonStateClass = "bg-slate-50 border-slate-200 text-slate-400 opacity-40 scale-95";
                                    }
                                }

                                return (
                                    <button
                                        key={idx}
                                        disabled={hasAnsweredCurrent}
                                        onClick={() => handleAnswerSelect(option)}
                                        className={`p-6 rounded-2xl border-2 font-black text-xl text-left transition-all duration-200 flex items-center justify-between break-all ${buttonStateClass} ${!hasAnsweredCurrent && 'transform hover:-translate-y-1 hover:shadow-md active:scale-95'}`}
                                    >
                                        {option}
                                        {Icon && <span className="animate-in zoom-in">{Icon}</span>}
                                    </button>
                                );
                            })}
                        </div>
                    </div>
                );
            };

            // --- 화면 렌더링 4: 퀴즈 결과 ---
            const renderResults = () => {
                const correctCount = userAnswers.filter(a => a.isCorrect).length;
                const percentage = Math.round((correctCount / quizQuestions.length) * 100);

                return (
                    <div className="max-w-4xl mx-auto animate-in zoom-in-95 duration-500">
                        <div className="bg-gradient-to-b from-indigo-600 to-violet-800 p-10 sm:p-16 rounded-[2.5rem] shadow-xl text-center mb-10 relative overflow-hidden">
                            <div className="absolute top-0 right-0 -mt-10 -mr-10 opacity-10">
                                <Sparkles className="w-64 h-64 text-white" />
                            </div>
                            
                            <div className="inline-flex items-center justify-center w-28 h-28 sm:w-32 sm:h-32 rounded-full bg-white mb-8 shadow-lg ring-8 ring-white/20">
                                <span className="text-4xl sm:text-5xl font-black text-transparent bg-clip-text bg-gradient-to-br from-indigo-600 to-violet-600">{percentage}%</span>
                            </div>
                            <h2 className="text-3xl sm:text-4xl font-black text-white mb-4 tracking-tight">
                                {percentage >= 80 ? '훌륭합니다! 마스터하셨네요! 🎉' : percentage >= 50 ? '잘 하셨어요! 조금만 더! 👍' : '다시 한번 도전해볼까요? 💪'}
                            </h2>
                            <p className="text-indigo-100 text-xl font-medium">
                                총 {quizQuestions.length}문제 중 <strong className="text-white text-2xl mx-1">{correctCount}문제</strong>를 맞혔습니다.
                            </p>
                            
                            <div className="mt-10 flex flex-col sm:flex-row justify-center gap-4 relative z-10">
                                <button onClick={startQuiz} className="px-8 py-4 bg-white text-indigo-700 hover:bg-indigo-50 rounded-2xl font-black text-lg transition-all shadow-md hover:shadow-lg flex items-center justify-center gap-2">
                                    <RotateCcw className="w-5 h-5" /> 다시 풀기
                                </button>
                                <button onClick={goHome} className="px-8 py-4 bg-indigo-900/40 hover:bg-indigo-900/60 border border-indigo-400/30 text-white rounded-2xl font-bold text-lg transition-all flex items-center justify-center gap-2">
                                    <Home className="w-5 h-5" /> 홈으로 이동
                                </button>
                            </div>
                        </div>

                        <div className="space-y-6">
                            <h3 className="font-black text-2xl text-slate-800 mb-6 px-2 flex items-center gap-3">
                                <BookOpen className="w-6 h-6 text-indigo-500" /> 상세 결과 및 오답 노트
                            </h3>
                            {userAnswers.map((answer, idx) => {
                                const wordObj = words.find(w => w.word.toLowerCase() === answer.question.correctAnswer.toLowerCase());
                                const originalDef = wordObj ? wordObj.definition : 'Definition not found.';

                                return (
                                    <div key={idx} className={`p-6 sm:p-8 rounded-3xl border-2 transition-all hover:shadow-md ${answer.isCorrect ? 'bg-white border-green-100' : 'bg-red-50/30 border-red-200'}`}>
                                        <div className="flex flex-col sm:flex-row items-start gap-4 sm:gap-6">
                                            <div className="mt-1 flex-shrink-0">
                                                {answer.isCorrect ? <CheckCircle className="w-8 h-8 text-green-500 drop-shadow-sm" /> : <XCircle className="w-8 h-8 text-red-500 drop-shadow-sm" />}
                                            </div>
                                            <div className="flex-1 w-full">
                                                <p className="text-xl sm:text-2xl text-slate-800 font-bold mb-6 break-words leading-relaxed">
                                                    {answer.question.question}
                                                </p>
                                                <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                                                    <div className="bg-slate-50 p-4 rounded-2xl border border-slate-100">
                                                        <p className="text-xs font-bold text-slate-400 uppercase tracking-wider mb-1">정답 단어</p>
                                                        <p className="text-2xl font-black text-indigo-600">{answer.question.correctAnswer}</p>
                                                    </div>
                                                    {!answer.isCorrect && (
                                                        <div className="bg-red-50 p-4 rounded-2xl border border-red-100">
                                                            <p className="text-xs font-bold text-red-400 uppercase tracking-wider mb-1">선택한 오답</p>
                                                            <p className="text-xl font-bold text-red-700">{answer.selectedAnswer}</p>
                                                        </div>
                                                    )}
                                                </div>
                                                <div className="mt-4 bg-indigo-50 p-5 rounded-2xl border border-indigo-100">
                                                    <p className="text-xs font-bold text-indigo-400 uppercase tracking-wider mb-2">단어 정의 (Definition)</p>
                                                    <p className="text-lg font-medium text-slate-800">{originalDef}</p>
                                                </div>
                                            </div>
                                        </div>
                                    </div>
                                );
                            })}
                        </div>
                    </div>
                );
            };

            return (
                <div className="pb-24 bg-slate-50 min-h-screen">
                    {/* 상단 네비게이션 헤더 */}
                    <header className="bg-white border-b border-slate-200 sticky top-0 z-50 shadow-sm/50 backdrop-blur-md bg-white/80">
                        <div className="max-w-5xl mx-auto px-4 sm:px-6 lg:px-8 h-16 flex items-center justify-between">
                            <div className="flex items-center gap-3 cursor-pointer" onClick={goHome}>
                                <div className="bg-indigo-600 p-2 rounded-xl shadow-sm text-white">
                                    <Book className="w-5 h-5" />
                                </div>
                                <h1 className="text-xl font-black text-slate-800 hidden sm:block tracking-tight">
                                    Vocab Master
                                </h1>
                            </div>
                            <div className="flex space-x-1 bg-slate-100 p-1.5 rounded-xl">
                                <button onClick={goHome} className={`px-4 py-1.5 rounded-lg text-sm font-bold transition-all ${activeTab === 'home' ? 'bg-white text-indigo-700 shadow-sm' : 'text-slate-500 hover:text-slate-800'}`}>홈(설정)</button>
                                <button disabled className={`px-4 py-1.5 rounded-lg text-sm font-bold transition-all ${activeTab === 'study' ? 'bg-white text-indigo-700 shadow-sm' : 'text-slate-400 opacity-50'}`}>학습장</button>
                                <button disabled className={`px-4 py-1.5 rounded-lg text-sm font-bold transition-all ${(activeTab === 'quiz' || activeTab === 'results') ? 'bg-white text-indigo-700 shadow-sm' : 'text-slate-400 opacity-50'}`}>퀴즈 모드</button>
                            </div>
                        </div>
                    </header>
                    
                    <main className="max-w-5xl mx-auto px-4 sm:px-6 lg:px-8 py-8 sm:py-12">
                        {activeTab === 'home' && renderHome()}
                        {activeTab === 'study' && renderStudy()}
                        {activeTab === 'quiz' && renderQuiz()}
                        {activeTab === 'results' && renderResults()}
                    </main>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
